tell application "System Events"		tell appearance preferences				if (dark mode) then			set dark mode to not dark mode			tell application "System Events"				tell every desktop					(* light mode wallpaper *)					set picture to "/path/to/wallpaper"				end tell			end tell		else			set dark mode to not dark mode			tell application "System Events"				tell every desktop					(* dark mode wallpaper *)					set picture to "/path/to/wallpaper"				end tell			end tell		end if			end tell	end tell
